防抖函数
debounce(func, delay) {
  let timer = null;
  return function(...args) {
    if(timer) {
      clearTimeout(timer)
    };
    timer = setTimeout(() => {
      func.apply(this, args)
    }, delay)
  }
}

节流函数
throttle(func, delay) {
  let timer = null;
  return function(...args) {
    if(!timer){
      timer = setTimeout(() =>{
        func.apply(this, args);
        timer = null;
      }, delay)
    }
  }
}
